```python
"""AWS Lambda handler for the portfolio chat agent.

Receives user messages via POST, sends them to Claude Haiku with
portfolio context, and returns the response.
"""

import json
from pathlib import Path
from typing import Any

import anthropic

# --- Constants ---

MAX_MESSAGE_LENGTH = 1000
MODEL_ID = "claude-haiku-4-5-20251001"
MAX_TOKENS = 512
ALLOWED_ORIGIN = "https://charleslikesdata.com"

# --- Knowledge Base (loaded once at cold start) ---

_knowledge_base: dict | None = None


def load_knowledge_base() -> dict:
    """Load the portfolio knowledge base from the bundled JSON file."""
    global _knowledge_base
    if _knowledge_base is None:
        kb_path = Path(__file__).parent / "knowledge_base.json"
        _knowledge_base = json.loads(kb_path.read_text())
    return _knowledge_base


def build_system_prompt() -> str:
    """Construct the system prompt from the knowledge base.

    Returns
    -------
    str
        System prompt giving Claude full context about Charles's
        portfolio, projects, skills, testimonials, and contact info.
    """
    kb = load_knowledge_base()
    person = kb["person"]
    skills = kb["skills"]
    projects = kb["projects"]
    testimonials = kb["testimonials"]

    # Build project summaries (include details when available)
    project_lines = []
    for i, p in enumerate(projects, 1):
        tags = ", ".join(p["tags"])
        featured = " [FEATURED]" if p.get("featured") else ""
        entry = (
            f"{i}. {p['title']}{featured}\n"
            f"   Tags: {tags}\n"
            f"   Description: {p['description']}\n"
            f"   URL: {p['url']}"
        )
        if p.get("details"):
            entry += f"\n   Details: {p['details']}"
        project_lines.append(entry)

    # Build testimonial summaries
    testimonial_lines = []
    for t in testimonials:
        testimonial_lines.append(
            f'- "{t["quote"]}" -- {t["author"]}, {t["title"]}, {t["company"]}'
        )

    # Flatten all skills
    all_skills = []
    for items in skills.values():
        all_skills.extend(items)

    return (
        f"You are a helpful assistant on the portfolio website of {person['name']}, "
        f"who is a {person['title']}. You answer questions about Charles, his projects, "
        f"skills, and experience in a friendly, professional, and concise tone.\n\n"
        f"## About {person['name']}\n"
        f"{person['bio']}\n"
        + (f"\n{person['extended_bio']}\n\n" if person.get("extended_bio") else "\n")
        f"## Contact\n"
        f"- Email: {person['email']}\n"
        f"- LinkedIn: {person['linkedin']}\n"
        f"- GitHub: {person['github']}\n"
        f"- Website: {person['website']}\n\n"
        f"## Skills\n"
        f"{', '.join(all_skills)}\n\n"
        f"## Projects ({len(projects)} total)\n"
        + "\n".join(project_lines)
        + "\n\n"
        f"## Testimonials from Colleagues\n"
        + "\n".join(testimonial_lines)
        + "\n\n"
        "## Instructions\n"
        "- Answer questions about Charles's projects, skills, background, and experience.\n"
        "- When referencing a project, include its URL so the user can explore further.\n"
        "- If the user asks something outside the scope of this portfolio, politely redirect.\n"
        "- Keep responses concise (2-4 sentences) unless the user asks for detail.\n"
        "- Be friendly and professional.\n"
        "- Do not make up information that is not in the knowledge base above.\n"
    )


# --- Anthropic Client ---


def get_anthropic_client() -> anthropic.Anthropic:
    """Create an Anthropic client using the ANTHROPIC_API_KEY env var."""
    return anthropic.Anthropic()


# --- Request Parsing ---


def parse_request(body: str | None) -> str:
    """Parse and validate the incoming request body.

    Parameters
    ----------
    body : str or None
        Raw JSON string from the Lambda event.

    Returns
    -------
    str
        The validated user message.

    Raises
    ------
    ValueError
        If body is missing, invalid JSON, missing 'message' key,
        or message is empty or too long.
    """
    if not body:
        raise ValueError("Request body is empty")
    try:
        data = json.loads(body)
    except json.JSONDecodeError as exc:
        raise ValueError("Invalid JSON in request body") from exc
    if "message" not in data:
        raise ValueError("Missing 'message' key in request body")
    message = data["message"].strip()
    if not message:
        raise ValueError("Message is empty")
    if len(message) > MAX_MESSAGE_LENGTH:
        raise ValueError(f"Message too long (max {MAX_MESSAGE_LENGTH} characters)")
    return message


# --- Response Building ---


def build_response(status_code: int, body: dict[str, Any]) -> dict[str, Any]:
    """Build a Lambda Function URL response with CORS headers.

    Parameters
    ----------
    status_code : int
        HTTP status code.
    body : dict
        Response body to serialize as JSON.

    Returns
    -------
    dict
        Lambda Function URL response object.
    """
    return {
        "statusCode": status_code,
        "headers": {
            "Content-Type": "application/json",
            "Access-Control-Allow-Origin": ALLOWED_ORIGIN,
            "Access-Control-Allow-Methods": "POST, OPTIONS",
            "Access-Control-Allow-Headers": "Content-Type",
        },
        "body": json.dumps(body),
    }


# --- Lambda Handler ---


def handler(event: dict, context: Any) -> dict[str, Any]:
    """Lambda Function URL handler.

    Parameters
    ----------
    event : dict
        Lambda Function URL event object.
    context : Any
        Lambda context (unused).

    Returns
    -------
    dict
        HTTP response with status code, headers, and body.
    """
    method = event.get("requestContext", {}).get("http", {}).get("method", "")

    # CORS preflight
    if method == "OPTIONS":
        return build_response(200, {})

    # Only allow POST
    if method != "POST":
        return build_response(405, {"error": "Method not allowed"})

    # Parse and validate request
    try:
        message = parse_request(event.get("body"))
    except ValueError as e:
        return build_response(400, {"error": str(e)})

    # Call Claude
    try:
        client = get_anthropic_client()
        response = client.messages.create(
            model=MODEL_ID,
            max_tokens=MAX_TOKENS,
            system=build_system_prompt(),
            messages=[{"role": "user", "content": message}],
        )
        assistant_text = response.content[0].text
        return build_response(200, {"response": assistant_text})
    except Exception:
        return build_response(500, {"error": "Internal server error"})
```

## Key Design Decisions

1. **Stateless requests:** Each Lambda invocation is independent. No conversation history is maintained server-side. This keeps the Lambda simple and avoids DynamoDB.

2. **Module-level caching:** `_knowledge_base` is loaded once per cold start and reused across warm invocations. This avoids reading the JSON file on every request.

3. **Mocked Anthropic client in tests:** The `TestHandler` class uses `pytest-mock` to patch `get_anthropic_client()`. This avoids real API calls in tests while verifying the full request/response flow.

4. **CORS hardcoded to production domain:** `ALLOWED_ORIGIN` is set to `https://charleslikesdata.com`. For local development, you may temporarily change this to `http://localhost:8000` or `*`.

5. **500 error catch-all:** The handler catches all exceptions from the Anthropic API call and returns a generic 500. This prevents leaking internal errors to the client.
